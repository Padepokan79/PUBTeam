//Write a query in SQL to find the teams played the first match of EURO cup 2016.
1)select country_name from soccer_country
where country_id in(select team_id from match_details
where match_no=1);

//Write a query in SQL to find the winner of EURO cup 2016
2)select country_name from soccer_country
where country_id in(select team_id from match_details
where win_lose='W');

//Write a query in SQL to find the players with other information under contract to Liverpool were in the Squad of England in 2016 EURO cup
