//menampilkan nip, nama pegawai

//menampilkan nip, nama, TMTGAJI yang TMTGAJI 2011-01-01

//menampilkan nip yang TMTGAJI-nya paling pertama
SELECT NIP, MIN(TMTGAJI) AS Pegawai_Paling_Lama FROM historis_fungsikhusus;
