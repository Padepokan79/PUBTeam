1. Menampilkan nama negara bertanding pertama

SELECT country_name
FROM soccer_country
WHERE country_id IN
(SELECT team_id
FROM match_details
where match_no=1)


2. Mencari juara piala dunia
SELECT country_name
FROM soccer_country
WHERE country_id  IN(
   SELECT team_id 
   FROM match_details
   WHERE win_lose = 'W' 
   AND play_stage ='F'
);

3. Menampilkan Penonton Terbanyak
SELECT match_no ,play_stage,goal_score ,audence 
FROM match_mast
WHERE audence = (SELECT MAX(audence) FROM match_mast)

4. Menampilkan nomer pertandingan germany vs poland
SELECT DISTINCT match_no as nomor
FROM match_details
WHERE team_id IN (
  SELECT country_id 
  FROM soccer_country 
  WHERE country_name IN(
     'Germany','Poland'
  )
  and win_lose IN(
     'D'
  )
)
