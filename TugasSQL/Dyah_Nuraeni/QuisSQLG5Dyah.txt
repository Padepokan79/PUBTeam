17.menampilkan kepala sekolah yng memiliki gaji tertinggi,islam,tidak memiliki eselon menikah,lk,pembina
--------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
select nama from mstpegawai where tjeselon =0 and agama in (
select kdagama from agama_tbl where nmagama = "islam") and kdstawin in ( 
select kd_stawin from tb_stawin where nm_stawin = "nikah") and kdjenkel = 1 and kdpangkat in(
select kdpangkat from pangkat_tbl where nmpangkat = "pembina" ) group by nama having(max(gapok));
