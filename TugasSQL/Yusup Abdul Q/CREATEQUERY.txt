
1.HALAH//MENCARI NAMA PEGAWAI YANG  JUMLAH HUTANG TERTINGGI DAN JUMLAH CICILAN DI ATAS RATA-RATA DAN PANGKATNYA DUTA BESAR
SELECT NAMA
FROM MSTPEGAWAI WHERE NIP IN(
SELECT NIP FROM HUTANG 
WHERE JMLHUTANG IN(
SELECT MAX(JMLHUTANG) 
FROM HUTANG WHERE JMLCICILAN >(
SELECT AVG(JMLCICILAN) 
FROM HUTANG WHERE KETERANGAN IS NULL))) 
&& KDPANGKAT IN(
SELECT KDPANGKAT 
FROM PANGKAT_TBL WHERE NMPANGKAT ='DUTA BESAR') 

2.TEGUH//MENCARI NAMA DAN TANGGAL WAFAT YANG MEMILKI GAJI POKOK LEBIH DARI 3000000
SELECT FB.NAMA, FU.TGLWAFAT 
FROM FGAJI_UANGDUKA FU 
INNER JOIN FGAJI_BARU FB 
ON FU.NIP = FB.NIP 
WHERE FU.GAPOK> 3000000 
GROUP BY FB.NAMA

6.FRIDA//MENCARI NAM,GOLONGAN,PANGKAT YANG GAJI POKOKNYA DIATAS RATA-RATA DAN MASA KERJA LEBIH DARI 30 TAHUN
SELECT M.NAMA,P.NMGOL,P.NMPANGKAT 
FROM MSTPEGAWAI M 
INNER JOIN PANGKAT_TBL P 
ON M.KDPANGKAT =P.KDPANGKAT 
WHERE M.GAPOK >(
SELECT AVG(GAPOK) 
ROM MSTPEGAWAI 
WHERE MASKER >30)

8.JEJEN//MENCARI NAMA PEGAWAI YANG BERAGAMA ISLAM, JENIS KELAMIN PEREMPUAN YANG JABATAN GURUNYA SEBAGAI KEPALA SEKOLAH SMP GOL II YAN MASKER NYA 24 DAN MEMPUMPUNYAI GELAR DRA SERTA KELAHIRAN 1963
SELECT M.NAMA FROM MSTPEGAWAI M INNER JOIN AGAMA_TBL A 
INNER JOIN GURU_TBL G 
ON M.AGAMA =A.KDAGAMA 
AND M.KDGURU =G.KDGURU 
WHERE A.NMAGAMA='ISLAM' 
AND M.KDJENKEL =2 
AND G.NMGURU ='KEPALA SEKOLAH SMP.GOL.II'
AND M.MASKER =24 
AND M.NAMA LIKE 'DRA%'
AND M.TGLLHR LIKE '%63%';

9.IPIN//MENCARI NAMA ,GAJI POKOK, PANGKAT YANG MEMILIKI HUTANG TERTINGGI

WHERE H.JMLHUTANG IN(
SELECT MAX(JMLHUTANG) 
FROM HUTANG)

12.NATIA//MENCARI NI,NAMA,JUMLAH DARI NILAI ATAS YANG KD PANGKAT 3A GAPOK > 2JT JUMLAH ISTRI 1DAN JUMLAH ANAK 2
SELECT D.NIP,D.NAMA,COUNT(G.NILAI_ATAS) 
FROM DATA_RAPEL D 
INNER JOIN GAPOK_TBL G
WHERE G.KDPANGKAT ='3A' 
AND D.GAPOK>2000000 
AND D.JISTRI =1 
AND JANAK=2;


13.TEGAR//MENCARI PEGAWAI YANG MEMILIKI PANGKAT WALIKOTA. BUPATI/WALIKOTA YANG BERGELAR HAJI DAN MEMILIKI ISTRI 4
SELECT F.NAMA 
FROM FGAJI F 
INNER JOIN PANGKAT_TBL P
WHERE P.NMPANGKAT ='WAKIL. BUPATI/WALIKOTA' 
AND(F.NAMA LIKE '%H.%' 
OR F.NAMA LIKE '%H,%'
OR F.GLRDEPAN ='H,' 
OR F.GLRDEPAN ='H') 
AND F.JISTRI =4
LIMIT 10;

14.RESTI//MENAMPILKAN NAMA PEGAWAI YANG POTONGAN PALING KECIL DAN JUMLAH ANAK TERBANYAK DAN AGAMA ISLAM

SELECT F.NAMA 
FROM MSTPEGAWAI M 
INNER JOIN AGAMA_TBL A
INNER JOIN FGAJI F
ON M.AGAMA =A.KDAGAMA 
AND M.NIP = F.NIP 
WHERE F.POTONGAN IN(
SELECT MIN(POTONGAN) 
FROM FGAJI 
WHERE JANAK IN(
SELECT MAX(JANAK) 
FROM FGAJI)) AND A.NMAGAMA ='ISLAM' LIMIT 10;
