1//MENCARI JUMLAH STADION ATAU TEMPAT PADA eURO 2016
SELECT COUNT(*) FROM SOCCER_VENUE

2//MENCARI JUMLAH NEGARA YANG BERPARTISIPASI DALAM UERO 2016
SELECT DISTINCT COUNT(TEAM_ID) FROM PLAYER_MAST GROUP BY TEAM_ID

3//MENCARI JUMLAH GOAL PADA WAKTU NORMAL PADA EURO 2016
SELECT COUNT(GOAL_ID) FROM GOAL_DETAILS WHERE GOAL_TIME <= 90 

4//MENCARI JUMLAH PERTANDINGAN HASIL NYA ADA PEMENANGNYA PADA EURO 2016
SELECT COUNT(RESULTS) FROM MATCH_MAST WHERE RESULTS = 'WIN'

5//MENCARI PERTANDINGAN DENGAN HASIL AKHIR DRAW ATAU IMBANG PADA EURO 2016
SELECT COUNT(RESULTS) FROM MATCH_MAST WHERE RESULTS = 'DRAW'

6//MENCARI TANGGAL DIMULAINYA PERTANDINGAN PADA EURO 2016
SELECT PLAY_DATE FROM MATCH_MAST WHERE MATCH_NO = 1 

7//MENCARI JUMLAH GOL BUNUH DIRI PADA EURO 2016
SELECT COUNT(GOAL_TYPE) FROM GOAL_DETAILS WHERE GOAL_TYPE = 'O' 

8//MENCARI JUMLAH PERTANDINGAN DENGAN HASIL ADA PEMENANGNYA DI PENYISIHAN GROUP PADA EURO 2016
SELECT COUNT(MATCH_NO) FROM MATCH_MAST WHERE RESULTS = 'WIN' AND PLAY_STAGE ='G' GROUP BY PLAY_STAGE 


9//MENETUKAN YANG DI TENTUKAN DENGAN PENALTI
SELECT COUNT(*) FROM PENALTY_SHOOTOUT


10//MENCARI JUMLAH PERTANDINGAN YANG DI BABAK 16 BESAR YANG DI TENTUKAN DENGAN ADU PENALTI PADA EURO 2016
SELECT COUNT(MATCH_NO) FROM MATCH_MAST WHERE PLAY_STAGE = 'R' AND  DECIDED_BY ='P';


11//MENCARI JUMLAH PERTANDINGAN DENGAN HASIL DRAW DI PENYISISAN GROUP PADA EURO 2016 
SELECT COUNT(MATCH_NO) FROM MATCH_DETAILS WHERE WIN_LOSE ='D' AND PLAY_STAGE ='G' GROUP BY PLAY_STAGE

12//MENAMPILKAN NO PERTANDINGA,TANGGAL PERTANDINGAN, DAN JUMLAH GOL YANG DI CETAK DI BABAK PERTAMA PERTAMBAHAN WAKTU PADA EURO 2016 
SELECT MM.MATCH_NO,MM.PLAY_DATE,COUNT(GS.GOAL_TIME)
FROM GOAL_DETAILS GS,MATCH_MAST MM 
WHERE MM.MATCH_NO = GS.MATCH_NO 
AND GS.GOAL_HALF ='1' 
AND GS.GOAL_SCHEDULE ='ST' 
GROUP BY MM.MATCH_NO


13//MENAMPILKAN JUMLAH PERTANDING DENGAN HASIL DRAW DENGAN SCORE 0 PADA EURO 2016 
SELECT COUNT(MATCH_NO) FROM MATCH_DETAILS WHERE WIN_LOSE ='D' && GOAL_SCORE= 0  

14//MENCARI JUMLAH PERTANDINGAN DENGAN GOAL 1 DAN BUKAN DARI HASIL TENDANGAN PENALTY PADA EURO 2016
SELECT COUNT(MATCH_NO) FROM MATCH_DETAILS WHERE GOAL_SCORE = 1 AND DECIDED_BY <> 'P' 


15//MENCARI JUMLAH PEMAIN YANG DI GANTI DI SETIA PERTADINGAN PADA EURO 2016
SELECT COUNT(PLAYER_ID) FROM PLAYER_IN_OUT WHERE IN_OUT= 'O'

16//MENCARI JUMLAH PEMAIN YANG DIMASUKAN DI JAM NORMAL PADA EURO 2016
SELECT COUNT(PLAYER_ID) FROM PLAYER_IN_OUT WHERE IN_OUT= 'I' AND TIME_IN_OUT <=90 AND PLAY_SCHEDULE ='NT'


17//MENCARI JUMLAH PEMAIN YANG DIGANTI PAS STOP TIME  PADA EURO 2016
SELECT COUNT(PLAYER_ID) FROM PLAYER_IN_OUT 
WHERE PLAY_SCHEDULE ='ST' && IN_OUT= 'O' 

18//MENCARI JUMLAH PEMAIN YANG HAFL TIME BABAK PERTAMA
SELECT COUNT(PLAYER_ID) FROM PLAYER_IN_OUT 
WHERE PLAY_HALF = 1 && IN_OUT= 'O' 

19//JUMLAH GOAL DARI PERTANDINGAN IMBANG
SELECT SUM(GOAL_SCORE) FROM MATCH_DETAILS WHERE WIN_LOSE ='D'

20//JUMLAG YANG DI GANTI PADA TAMBAHAN WAKTU
SELECT COUNT(PLAYER_ID) FROM PLAYER_IN_OUT WHERE PLAY_SCHEDULE ='ET' && IN_OUT= 'O' 

21//
SELECT PLAY_SCHEDULE,COUNT(IN_OUT) FROM PLAYER_IN_OUT GROUP BY PLAY_SCHEDULE


22//JUMLAH TENDANGAN PENALTY
SELECT COUNT(*) FROM PENALTY_SHOOTOUT 

23//JUMLAH TENDANGAN PINALTY YANG GOLLLLLLLLLLL
SELECT COUNT(*) FROM PENALTY_SHOOTOUT WHERE SCORE_GOAL ='Y' 

24//JUMLAH PENALTI YANG GAGAL 
SELECT COUNT(*) FROM PENALTY_SHOOTOUT WHERE SCORE_GOAL ='N' 

25//MENAMPILKAN NAMA PEMAIN DAN JUMLAH TENDANGAN PENALTI
SELECT PM.PLAYER_NAME,COUNT(PS.KICK_ID) AS JUMLAH_TENDANGAN 
FROM PENALTY_SHOOTOUT PS 
INNER JOIN PLAYER_MAST PM 
ON PM.PLAYER_ID = PS.PLAYER_ID 
GROUP BY PM.PLAYER_ID ORDER BY PM.PLAYER_NAME  


26//
SELECT SC.COUNTRY_NAME,COUNT(PS.TEAM_ID) AS JUMLAH_TENDANGAN_PENALTI  
FROM PENALTY_SHOOTOUT PS 
INNER JOIN SOCCER_COUNTRY SC
ON SC.COUNTRY_ID = PS.TEAM_ID 
GROUP BY SC.COUNTRY_NAME
