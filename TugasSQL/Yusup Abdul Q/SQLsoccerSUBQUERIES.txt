CREATE BY YUSP ABDUL QOHAR 

SUBSELECT 

1//MENCARI NAMA NEGARA YANG BERTANDING PERTAMA PADA EURO 2016
SELECT SC.COUNTRY_NAME 
FROM SOCCER_COUNTRY SC,MATCH_DETAILS MD 
WHERE SC.COUNTRY_ID =MD.TEAM_ID 
AND MD.MATCH_NO = 1 

2//MENCARI NAMA NEGARA YANG JUARA PADA EURO 2016
SELECT SC.COUNTRY_NAME 
FROM SOCCER_COUNTRY SC 
INNER JOIN MATCH_DETAILS MD 
ON SC.COUNTRY_ID =MD.TEAM_ID WHERE WIN_LOSE ='W' AND MD.PLAY_STAGE = 'F'

3//MENCARI MATCH_NO, PLAY_STAGE,GOAL_SCORE, DAN AUDENCE YANG MEMILIKI AUDENCE YANG TERTINGGI PADA EURO 2016  
SELECT match_no ,PLAY_STAGE,GOAL_SCORE,AUDENCE AS AUDENCE_TERBANYAK  
FROM MATCH_MAST 
WHERE AUDENCE 
IN (SELECT MAX(AUDENCE) FROM MATCH_MAST)

4//MENCARI MATCH_NO ANTARA GERMANY DAN POLAND PADA EURO 2017
SELECT MATCH_NO FROM MATCH_DETAILS 
WHERE TEAM_ID IN(
SELECT COUNTRY_ID 
FROM SOCCER_COUNTRY 
WHERE COUNTRY_ID = 1208  
AND COUNTRY_ID = 1213) 

5//
SELECT MATCH_NO,PLAY_STAGE,PLAY_DATE,GOAL_SCORE,RESULTS 
FROM MATCH_DETAILS 
WHERE MATCH_NO IN(
SELECT MATCH_NO FROM MATCH_MAST 
WHERE TEAM_ID IN(
SELECT COUNTRY_ID FROM SOCCER_COUNTRY 
WHERE COUNTRY_ID = 1209 OR COUNTRY_ID = 1214))
