FROM SOCCER_COUNTRY SC,MATCH_DETAILS MD 
WHERE SC.COUNTRY_ID =MD.TEAM_ID 
AND MD.MATCH_NO = 1 

2//MENCARI NAMA NEGARA YANG JUARA PADA EURO 2016
SELECT SC.COUNTRY_NAME 
FROM SOCCER_COUNTRY SC 
INNER JOIN MATCH_DETAILS MD 
ON SC.COUNTRY_ID =MD.TEAM_ID WHERE WIN_LOSE ='W' AND MD.PLAY_STAGE = 'F'

3//MENCARI MATCH_NO, PLAY_STAGE,GOAL_SCORE, DAN AUDENCE YANG MEMILIKI AUDENCE YANG TERTINGGI PADA EURO 2016  
SELECT match_no ,PLAY_STAGE,GOAL_SCORE,AUDENCE AS AUDENCE_TERBANYAK  
FROM MATCH_MAST 
WHERE AUDENCE 
IN (SELECT MAX(AUDENCE) FROM MATCH_MAST)

WHERE TEAM_ID IN(

FROM SOCCER_COUNTRY 
WHERE COUNTRY_ID = 1208  
AND COUNTRY_ID = 1213)

16//MENCARI NAMA PENJAGA GAWANG DARI ITALIA PADA EURO 2016
SELECT PLAYER_NAME 
FROM PLAYER_MAST

FROM PENALTY_GK 
WHERE TEAM_ID IN(

FROM SOCCER_COUNTRY 
WHERE COUNTRY_NAME ='Italy') )

SELECT COUNT(GOAL_ID) 
FROM GOAL_DETAILS

FROM SOCCER_COUNTRY 
WHERE COUNTRY_NAME ='Germany')

18//MENAMPILKAN NO_JERSEY,NAMA PEMAIN, POSISI SEBAGAI PENJAGA GAWANG PADA TEAM ENLISH PADA EURO 2016

FROM PLAYER_MAST

WHERE country_name ='England') 
&& POSI_TO_PLAY ='GK'

20//MENCARI NAMA PEMAIN YANG MENCETAK GOLA TERAKHIR PADA BABAK SEMI FINAL KEDUA PADA EURO 2016
SELECT PLAYER_NAME 
FROM PLAYER_MAST 
WHERE PLAYER_ID IN(

21//MENAMPILKAN NAMA CAPTAIN DARI WINNING PADA EURO 2016
SELECT PLAYER_NAME FROM PLAYER_MAST 
WHERE PLAYER_ID IN(
SELECT PLAYER_CAPTAIN 
FROM MATCH_CAPTAIN 
WHERE TEAM_ID IN(
SELECT TEAM_ID FROM MATCH_DETAILS 
WHERE PLAY_STAGE='F' 
&& WIN_LOSE = 'W'))

22//PADA EURO 2016
SELECT TEAM_ID, COUNT(TEAM_ID) 
FROM PLAYER_IN_OUT WHERE MATCH_NO IN (
SELECT MATCH_NO FROM MATCH_MAST 
WHERE PLAY_STAGE = 'F' && TEAM_ID IN(
SELECT COUNTRY_ID 
FROM SOCCER_COUNTRY 
WHERE COUNTRY_ID= 1207)) 
GROUP BY TEAM_ID //BLM BERES


23//MENCARI PENJAGA GAWAI DARI TEAM GERMANY YANG TIDAK KEBOBOLAN DI PENYISIHAN GROUP PADA EURO 2016
SELECT PLAYER_ID FROM PLAYER_MAST WHERE TEAM_ID IN(
SELECT TEAM_ID FROM MATCH_DETAILS WHERE PLAY_STAGE ='G' AND TEAM_ID IN(
SELECT COUNTRY_ID FROM SOCCER_COUNTRY WHERE COUNTRY_ID =1208))
AND POSI_TO_PLAY ='GK'//BLM BERES 

24//MENCARI RUNNER-UP PADA EURO 2016
SELECT SC.COUNTRY_NAME 
FROM SOCCER_COUNTRY SC 
INNER JOIN MATCH_DETAILS MD 
ON SC.COUNTRY_ID =MD.TEAM_ID 
WHERE MD.PLAY_STAGE ='F' AND MD.WIN_LOSE ='L'

25//MENCARI JUMLAH TENDANGAN PINALTI YANG TERBANYAK DI LAKUKAN OLEH TIAP TEAM PADA EURO 2016
SELECT MAX(KICK_NO) 
FROM PENALTY_SHOOTOUT 
WHERE KICK_NO IN(
SELECT COUNT(KICK_NO) AS JUMLAH_MAX 
FROM PENALTY_SHOOTOUT 
WHERE TEAM_ID IN(
SELECT COUNTRY_ID 
FROM SOCCER_COUNTRY) GROUP BY TEAM_ID) 

26//MENCARI JUMLAH MAXIMUM TEMBAKAN PENALTI YNAG DI LAKUKAN OLEH SETIAP PEMAIN PADA EURO 2016
SELECT MAX(KICK_ID) 
FROM PENALTY_SHOOTOUT 
WHERE PLAYER_ID IN(
SELECT PLAYER_ID 
FROM PLAYER_MAST 
WHERE TEAM_ID IN (
SELECT TEAM_ID FROM SOCCER_COUNTRY)) 

27//MENCARI MATCH_NO YANG MEMILIKI TENDANAN PENALTI TEBANYAK PADA EURO 2016
SELECT MATCH_NO 
FROM PENALTY_SHOOTOUT 
WHERE TEAM_ID IN(
SELECT MAX(TEAM_ID) 
FROM PENALTY_SHOOTOUT) 
GROUP BY MATCH_NO

29//MENCARI PEMAIN PORTUGAL YANG MENENDANG KE 7 SAAT ADU PENALTY MELAWAN POLAND PADA EURO 2016
SELECT PLAYER_ID 
FROM PENALTY_SHOOTOUT 
WHERE KICK_NO = 7 
AND TEAM_ID IN(
SELECT COUNTRY_ID 
FROM SOCCER_COUNTRY 
WHERE COUNTRY_ID = 1214) 
AND MATCH_NO IN(
SELECT MATCH_NO 
FROM PENALTY_SHOOTOUT)

30//MENCARI PLAY_STAGE DIMANA TENDANGAN PENALTI KE 23 DI TENDANG PADA EURO 2016
SELECT PLAY_STAGE 
FROM MATCH_MAST 
WHERE MATCH_NO IN(
SELECT MATCH_NO 
FROM PENALTY_SHOOTOUT 
WHERE KICK_ID = 23) 

31//MENAMPILKAN TEMPAT BERLANGSUNGNYA PERTANDINGAN YANG DILANJUTKAN DENGAN ADU PENALTI PADA EURO 2016
SELECT VENUE_NAME 
FROM SOCCER_VENUE 
WHERE VENUE_ID IN(
SELECT VENUE_ID FROM MATCH_MAST 
WHERE MATCH_NO IN(
SELECT MATCH_NO 
FROM PENALTY_SHOOTOUT))

32//MENCARI TANGGAGL PERTANDINGAN YANG DITENTUKAN DENGAN ADU PENALTI PADA EURO 2016
SELECT PLAY_DATE 
FROM MATCH_MAST 
WHERE MATCH_NO IN 
(SELECT MATCH_NO 
FROM PENALTY_SHOOTOUT)

33//MENCARI  7 GOAL TERCEPAT SETELAH 5 MENIT PADA EURO 2016
SELECT * FROM GOAL_DETAILS 
WHERE GOAL_TIME IN (
SELECT GOAL_TIME FROM GOAL_DETAILS WHERE GOAL_TIME > 5 ) 
ORDER BY GOAL_TIME 
LIMIT 7
