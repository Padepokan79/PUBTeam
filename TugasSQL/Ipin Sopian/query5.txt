//mencari nama, gaji, agama sama pangkat
1. select hasil.nama ,hasil.gapokb,hasil.maskerb,hasil.NMPANGKAT,a.nmagama from agama_tbl a inner join  ( select pg.nama ,fg.gapokb,fg.maskerb,NMPANGKAT,pg.AGAMA from pangkat_tbl pn left join mstpegawai pg right join fgaji_detil_ukp fg  on fg.NIP=pg.NIP on pn.KDPANGKAT=fg.KDPANGKATB  limit 10)as hasil on hasil.AGAMA=a.KDAGAMA where hasil.GAPOKB=(select max(gapokb) from fgaji_detil_ukp);
//mencari nama, gaji, agama sama pangkat dan pernah memiliki hutang
2. select hasil.nama ,hasil.gapokb,hasil.maskerb,hasil.NMPANGKAT,a.nmagama from agama_tbl a inner join  ( select pg.nama ,fg.gapokb,fg.maskerb,NMPANGKAT,pg.AGAMA from pangkat_tbl pn left join mstpegawai pg right join fgaji_detil_ukp fg  on fg.NIP=pg.NIP on pn.KDPANGKAT=fg.KDPANGKATB where pg.NIP in (select nip from hutang union select nip from mstpegawai) )as hasil on hasil.AGAMA=a.KDAGAMA ;
// mencari jumlah orang yang berhutang dng penghasilan lbih dari 2500000
3. select count(*) as jml_berhutang from (select hasil.nama ,hasil.gapokb,hasil.maskerb,hasil.NMPANGKAT,a.nmagama from agama_tbl a inner join  ( select pg.nama ,fg.gapokb,fg.maskerb,NMPANGKAT,pg.AGAMA from pangkat_tbl pn left join mstpegawai pg right join fgaji_detil_ukp fg  on fg.NIP=pg.NIP on pn.KDPANGKAT=fg.KDPANGKATB where pg.NIP in (select nip from hutang union select nip from mstpegawai) )as hasil on hasil.AGAMA=a.KDAGAMA where hasil.GAPOKB>2500000) as data;
//mencari nama, gaji, agama sama pangkat dan pernah memiliki hutang dan telah lunas
4. select hasil.nama ,hasil.gapokb,hasil.maskerb,hasil.NMPANGKAT,a.nmagama from agama_tbl a inner join  ( select pg.nama ,fg.gapokb,fg.maskerb,NMPANGKAT,pg.AGAMA from pangkat_tbl pn left join mstpegawai pg right join fgaji_detil_ukp fg  on fg.NIP=pg.NIP on pn.KDPANGKAT=fg.KDPANGKATB where pg.NIP in (select nip from hutang where tgllunas <> null union select nip from mstpegawai) )as hasil on hasil.AGAMA=a.KDAGAMA where hasil.GAPOKB>2500000;
//mencari jumlah orang yang berhutang dng penghasilan lbih dari 2500000 dan telah lunas
5. select count(*) as jml_berhutang from (select hasil.nama ,hasil.gapokb,hasil.maskerb,hasil.NMPANGKAT,a.nmagama from agama_tbl a inner join  ( select pg.nama ,fg.gapokb,fg.maskerb,NMPANGKAT,pg.AGAMA from pangkat_tbl pn left join mstpegawai pg right join fgaji_detil_ukp fg  on fg.NIP=pg.NIP on pn.KDPANGKAT=fg.KDPANGKATB where pg.NIP in (select nip from mstpegawai union select nip from hutang where tgllunas is not null ) )as hasil on hasil.AGAMA=a.KDAGAMA where hasil.GAPOKB>2500000) as data;
