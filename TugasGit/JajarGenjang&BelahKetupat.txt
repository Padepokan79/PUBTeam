//Keliling Belah Ketupat 	By : Cucu Yuni Yulainti

Luas = 1/2 x d1 x d2
Keliling = 4 x sisi
public class KelilingBelahKetupat{
	
	private int sisi;
	
	public KelilingBelahKetupat(int sisi){
		sisi = this.sisi;

	}
	
	public int hitung(){
		return 4*sisi;
	}

	
}


public Static void main(String args[]){
	KelilingBelahKetupat kel = new KelilingBelahKetupat(6);
	System.out.println("Keliling Belah Ketupat : " + kel.hitung);

}




//Luas Belah Ketupat
public class LuasBelahKetupat{
	
	private int diagonal1;
	private int diagonal2;
	
	public LuasBelahKetupat(int diagonal1, int diagonal2){
		diagonal1=this.diagonal1;
		diagonal2=this.diagonal2;

	}
public int hitungLuas(){
	return diagonal1*diagonal2*0,5;
}


public Static void main(String args[]){
	LuasBelahKetupat l = new KelilingBelahKetupat(6,5);
	System.out.println("Luas Belah Ketupat : " + l.hitungLuas);



}
